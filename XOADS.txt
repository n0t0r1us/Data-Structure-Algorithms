#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

// Chèn node vào cuối danh sách
void insertTail(Node*& head, int value) {
    Node* newNode = new Node;
    newNode->data = value;
    newNode->next = NULL;

    if (head == NULL) {
        head = newNode;
    } else {
        Node* current = head;
        while (current->next != NULL) {
            current = current->next;
        }
        current->next = newNode;
    }
}

// Xoá node có giá trị x
void deleteValue(Node*& head, int x) {
    if (head == NULL) return;

    // Nếu node đầu là x
    if (head->data == x) {
        Node* temp = head;
        head = head->next;
        delete temp;
        return;
    }

    Node* current = head;
    while (current->next != NULL && current->next->data != x) {
        current = current->next;
    }

    if (current->next != NULL) {
        Node* temp = current->next;
        current->next = current->next->next;
        delete temp;
    }
}

// In danh sách liên kết
void printList(Node* head) {
    Node* current = head;
    while (current != NULL) {
        cout << current->data << " ";
        current = current->next;
    }
}

int main() {
    int n, x;
    cin >> n >> x;

    Node* head = NULL;

    for (int i = 0; i < n; ++i) {
        int value;
        cin >> value;
        insertTail(head, value);
    }

    deleteValue(head, x); // Xóa giá trị x nếu có
    printList(head);      // In danh sách

    return 0;
}
