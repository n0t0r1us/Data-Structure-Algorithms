#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int T;
    cin >> T;
    while (T--) {
        int N, K;
        long long P;
        cin >> N >> K >> P;
        vector<long long> a(N);
        for (int i = 0; i < N; ++i) {
            cin >> a[i];
        }

        sort(a.begin(), a.end());

        int j = 0;
        bool found = false;
        for (int i = 0; i < N; ++i) {
            while (j < N && a[j] - a[i] <= 2 * P) {
                ++j;
            }
            if (j - i >= K) {
                found = true;
                break;
            }
        }

        if (found) cout << "NO\n";
        else cout << "YES\n";
    }
    return 0;
}
