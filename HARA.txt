#include <iostream>
#include <stack>
using namespace std;

int main() {
    int n;
    cin >> n;

    int a[n + 2]; // thêm phần tử cuối để xử lý nốt stack
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    a[n] = 0; // phần tử cuối để đẩy hết stack ra

    stack<int> st;
    int res = 0;

    for (int i = 0; i <= n; i++) {
        while (!st.empty() && st.top() > a[i]) {
            st.pop();
            res++;
        }
        if (st.empty() || st.top() < a[i]) {
            st.push(a[i]);
        }
        // nếu bằng nhau thì không cần làm gì
    }

    cout << res;
    return 0;
}
