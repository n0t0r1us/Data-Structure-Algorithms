#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    int a[100]; // giả sử n <= 100

    for (int i = 0; i < n; ++i)
        cin >> a[i];

    for (int i = 0; i < n - 1; ++i) {
        int minIndex = i;
        for (int j = i + 1; j < n; ++j)
            if (a[j] < a[minIndex])
                minIndex = j;
        // Đổi chỗ
        int tmp = a[i];
        a[i] = a[minIndex];
        a[minIndex] = tmp;
    }

    for (int i = 0; i < n; ++i)
        cout << a[i] << " ";

    return 0;
}
