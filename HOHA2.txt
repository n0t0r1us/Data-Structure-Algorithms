#include <iostream>
#include <cmath>
using namespace std;

typedef long long ll;

int main() {
    ll n;
    cin >> n;

    ll sum = 1; // 1 luôn là ước của mọi số >1

    for (ll i = 2; i * i <= n; ++i) {
        if (n % i == 0) {
            sum += i;
            ll j = n / i;
            if (i != j && j != n) sum += j;
        }
    }

    if (n == 1) sum = 0; // 1 không phải số hoàn hảo
    if (sum == n) cout << "YES\n";
    else cout << "NO\n";

    return 0;
}
