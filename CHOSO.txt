#include <iostream>
#include <map>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n; 
    cin >> n;
    int *A = new int[n];
    int *B = new int[n];
    int *C = new int[n];

    for (int i = 0; i < n; i++) cin >> A[i];
    for (int i = 0; i < n; i++) cin >> B[i];
    for (int i = 0; i < n; i++) cin >> C[i];

    map<int,int> freqA, freqB, freqC;

    for (int i = 0; i < n; i++) freqA[A[i]]++;
    for (int i = 0; i < n; i++) freqB[B[i]]++;
    for (int i = 0; i < n; i++) freqC[C[i]]++;

    long long res = 0;
    // Iterate over freqA keys, check if key in freqB and freqC
    for (map<int,int>::iterator it = freqA.begin(); it != freqA.end(); it++) {
        int x = it->first;
        if (freqB.count(x) && freqC.count(x)) {
            res += (long long)it->second * freqB[x] * freqC[x];
        }
    }

    cout << res << "\n";

    delete[] A;
    delete[] B;
    delete[] C;

    return 0;
}
