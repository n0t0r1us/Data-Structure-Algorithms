#include <iostream>
using namespace std;

int main() {
    int n, x;
    int a[100];

    cin >> n;
    for (int i = 0; i < n; ++i)
        cin >> a[i];

    cin >> x;

    // In vị trí xuất hiện của x
    for (int i = 0; i < n; ++i)
        if (a[i] == x)
            cout << i << " ";
    cout << endl;

    // Xóa x khỏi mảng
    int new_n = 0;
    for (int i = 0; i < n; ++i)
        if (a[i] != x)
            a[new_n++] = a[i];

    // In mảng sau khi xóa
    for (int i = 0; i < new_n; ++i)
        cout << a[i] << " ";

    return 0;
}
