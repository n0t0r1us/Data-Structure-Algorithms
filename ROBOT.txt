#include <iostream>
#include <vector>
using namespace std;

// Định nghĩa hàm getDir bên ngoài main
int getDir(pair<int,int> a, pair<int,int> b) {
    int dx = b.first - a.first;
    int dy = b.second - a.second;
    if (dx == 1 && dy == 0) return 1; // phải
    if (dx == -1 && dy == 0) return 3; // trái
    if (dx == 0 && dy == 1) return 0; // lên
    if (dx == 0 && dy == -1) return 2; // xuống
    return -1; // không hợp lệ
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    vector<pair<int,int> > p(n);
    for (int i = 0; i < n; i++) {
        cin >> p[i].first >> p[i].second;
    }

    int countRight = 0;
    for (int i = 1; i < n - 1; i++) {
        int d1 = getDir(p[i-1], p[i]);
        int d2 = getDir(p[i], p[i+1]);
        if (d1 != -1 && d2 != -1) {
            if ((d2 - d1 + 4) % 4 == 1) {
                countRight++;
            }
        }
    }
    cout << countRight << "\n";

    return 0;
}
