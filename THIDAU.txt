#include <iostream>

using namespace std;

void Optimize(int n,int f[],int a[], int b[],int k)
{
	f[1]=b[1];
	int ans=0;
	for (int i=2;i<=n;i++)
	   for (int j=1;j<=i;j++)
	   	if (a[i]-a[j]>=k)
	   	{
	   	f[i]=max(f[i],f[j]+b[i]);
	   	ans=max(ans,f[i]);
	   	}
	   	 else 
	   	 {
	   		f[i]=max(f[i],b[i]);
	   		ans=max(ans,f[i]);
	   		 break;
	   	}
	cout<<ans;
}

int main()
{
	int n,a[366],b[101],f[366],k;
	cin>>n>>k;
	for (int i=1;i<=n;i++) f[i]=0;
	for (int i=1;i<=n;i++) cin>>a[i];
	for (int i=1;i<=n;i++) cin>>b[i];
	Optimize(n,f,a,b,k);
}
