#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int N;
    cin >> N;
    vector<int> a(N);
    for (int i = 0; i < N; i++) cin >> a[i];
    ll res = 0;
    for (int b = 0; b < 21; b++) {
        ll cnt1 = 0;
        for (int i = 0; i < N; i++) {
            if (a[i] & (1 << b)) cnt1++;
        }
        ll cnt0 = N - cnt1;
        res += cnt1 * cnt0 * (1LL << b);
    }
    cout << res << "\n";
    return 0;
}
