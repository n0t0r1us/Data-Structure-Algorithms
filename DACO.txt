#include <iostream>

using namespace std;

void Optimize(int n,int a[],int f[],int l[])
{
    for (int i=1;i<=n-1;i++)
	   for (int j=i+1;j<=n;j++)
	     if (a[i]<a[j] && f[i]+1>f[j])
		 {
		 	l[j]=i;
		 	f[j]=f[i]+1;
		  }	
}

void Trace(int n,int a[],int b[],int f[],int l[])
{
	int k=0,vt,d=0;
	for (int i=1;i<=n;i++)
	   if (k<f[i])
	   {
	   	k=f[i];
	   	vt=i;
	   }
	while (f[vt] != 0)
	{
		d++;
		b[d]=a[vt];
		vt=l[vt];
	}
	cout<<d<<endl;;
	for (int i=d;i>=1;i--) 
	   cout<<b[i]<<" ";
}

int main()
{
	int n,a[1001],b[1001],f[1001],l[1001];
	cin>>n;
	for (int i=1;i<=n;i++) 
	{
	   cin>>a[i];
	   f[i]=1;
	   l[i]=0;
	}
   f[n+1]=INT_MAX;	
   Optimize(n,a,f,l);
   Trace(n,a,b,f,l);
}
