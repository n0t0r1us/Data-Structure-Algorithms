#include <iostream>
using namespace std;

bool isLeap(int year) {
    if (year % 400 == 0) return true;
    if (year % 100 == 0) return false;
    if (year % 4 == 0) return true;
    return false;
}

int main() {
    int d, t, n;
    cin >> d >> t >> n;

    // Số ngày mỗi tháng
    int days_in_month[] = {0,31,28,31,30,31,30,31,31,30,31,30,31};

    // Nếu năm nhuận thì tháng 2 có 29 ngày
    if (isLeap(n)) days_in_month[2] = 29;

    d++; // ngày tiếp theo

    if (d > days_in_month[t]) {
        d = 1;
        t++;
        if (t > 12) {
            t = 1;
            n++;
        }
    }

    cout << d << " " << t << " " << n << "\n";
    return 0;
}
