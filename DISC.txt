#include <iostream>
#include <list>
#include <map>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    list<int> disks;
    map<int, list<int>::iterator> position;

    // Khởi tạo danh sách đĩa và lưu vị trí iterator
    for (int i = 1; i <= n; ++i) {
        disks.push_back(i);
        list<int>::iterator it = disks.end();
        --it;
        position[i] = it;
    }

    for (int i = 0; i < m; ++i) {
        int x;
        cin >> x;

        // Xóa đĩa x khỏi vị trí cũ
        disks.erase(position[x]);

        // Đưa x lên đầu danh sách
        disks.push_front(x);
        position[x] = disks.begin();
    }

    // In ra thứ tự cuối cùng
    for (list<int>::iterator it = disks.begin(); it != disks.end(); ++it) {
        cout << *it << " ";
    }

    return 0;
}
