#include <iostream>
#include <vector>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    if (!(cin >> n)) return 0;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    long long res = 1; // smallest sum we cannot make so far
    for (int i = 0; i < n; ++i) {
        if (a[i] > res) break;
        res += a[i];
    }
    cout << res << "\n";
    return 0;
}
