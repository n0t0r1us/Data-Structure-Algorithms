#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    long long N;
    cin >> N;
    long long odd_count = 0, even_count = 0;
    for (long long i = 0; i < N; i++) {
        long long x;
        cin >> x;
        if (x % 2 == 0) even_count++;
        else odd_count++;
    }
    cout << odd_count * even_count << "\n";
    return 0;
}
