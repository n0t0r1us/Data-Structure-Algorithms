#include <iostream>
using namespace std;

int main() {
    int n, m, a[25];
    cin >> n >> m;
    for (int i = 0; i < n; ++i)
        cin >> a[i];

    for (int row = 0; row < m; ++row) {
        for (int col = 0; col < n; ++col) {
            if (a[col] >= m - row)
                cout << "#";
            else
                cout << ".";
        }
        cout << endl;
    }

    return 0;
}
