#include <iostream>
using namespace std;

void dichPhai1ViTri(int a[], int n) {
    int temp = a[n - 1]; // Luu ph?n t? cu?i
    for (int i = n - 1; i > 0; --i) {
        a[i] = a[i - 1]; // D?ch t?ng ph?n t? sang ph?i
    }
    a[0] = temp; // Ðua ph?n t? cu?i lên d?u
}

int main() {
    int n;
    cin >> n; // Nh?p s? ph?n t?

    int a[100];
    for (int i = 0; i < n; ++i)
        cin >> a[i]; // Nh?p m?ng

    dichPhai1ViTri(a, n); // G?i hàm d?ch

    for (int i = 0; i < n; ++i)
        cout << a[i] << " "; // In m?ng sau khi d?ch

    return 0;
}
