#include <iostream>
#include <stack>
#include <sstream>
#include <string>
#include <cctype>
using namespace std;

int doUuTien(char op) {
    if (op == '^') return 3;
    if (op == '*' || op == '/') return 2;
    if (op == '+' || op == '-') return 1;
    return 0;
}

bool laToanTu(char c) {
    return c == '+' || c == '-' || c == '*' || c == '/' || c == '^';
}

string trungToSangHauTo(string expr) {
    stack<char> st;
    stringstream ss(expr);
    string token, ketQua = "";

    while (ss >> token) {
        if (isdigit(token[0]) || (token.length() > 1 && token[0] == '-')) {
            ketQua += token + " ";
        }
        else if (token == "(") {
            st.push('(');
        }
        else if (token == ")") {
            while (!st.empty() && st.top() != '(') {
                ketQua += st.top();
                ketQua += " ";
                st.pop();
            }
            if (!st.empty() && st.top() == '(') st.pop();
        }
        else if (laToanTu(token[0])) {
            char op = token[0];
            while (!st.empty() && doUuTien(st.top()) >= doUuTien(op)) {
                ketQua += st.top();
                ketQua += " ";
                st.pop();
            }
            st.push(op);
        }
    }

    while (!st.empty()) {
        ketQua += st.top();
        ketQua += " ";
        st.pop();
    }

    return ketQua;
}

int main() {
    string expr;
    getline(cin, expr);
    cout << trungToSangHauTo(expr) << endl;
    return 0;
}
