#include <cstdio>
#include <cstdlib>

#define MOD 1000000007

int cmp(const void *x, const void *y) {
    long long a = *(const long long*)x;
    long long b = *(const long long*)y;
    if (a < b) return -1;
    else if (a > b) return 1;
    else return 0;
}

int main() {
    int N;
    std::scanf("%d", &N);
    long long *a = new long long[N];
    int i;
    long long max = 0;

    for (i = 0; i < N; i++) {
        std::scanf("%lld", &a[i]);
        if (a[i] > max) max = a[i];
    }

    qsort(a, N, sizeof(long long), cmp);

    long long sum = 0;
    for (i = 0; i < N; i++) {
        long long val = ((2LL * i - N + 1) * a[i]) % MOD;
        if (val < 0) val += MOD;
        sum = (sum + val) % MOD;
    }

    long long ans = (sum * (max % MOD)) % MOD;
    std::printf("%lld\n", ans);

    delete[] a;
    return 0;
}
