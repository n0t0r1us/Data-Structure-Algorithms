#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<int> stack;
    for (int i = 1; i <= n; ++i)
        stack.push_back(i);

    for (int i = 0; i < m; ++i) {
        int x;
        cin >> x;
        // Tìm và xóa đĩa x khỏi stack
        vector<int>::iterator it = find(stack.begin(), stack.end(), x);
        if (it != stack.end()) {
            stack.erase(it);
        }
        // Đưa đĩa x lên đầu stack
        stack.insert(stack.begin(), x);
    }

    for (int i = 0; i < (int)stack.size(); ++i)
        cout << stack[i] << " ";
    cout << endl;
    return 0;
}
