#include <iostream>
using namespace std;

void daoNguocMang(int a[], int n) {
    for (int i = 0; i < n / 2; ++i) {
        int temp = a[i];
        a[i] = a[n - 1 - i];
        a[n - 1 - i] = temp;
    }
}

int main() {
    int n;
    cin >> n;            // Nh?p s? ph?n t?
    int a[100];
    for (int i = 0; i < n; ++i)
        cin >> a[i];     // Nh?p m?ng

    daoNguocMang(a, n);  // √ê?o m?ng

    for (int i = 0; i < n; ++i)
        cout << a[i] << " ";  // In k?t qu?
    return 0;
}
