#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    long long n;
    if (!(cin >> n)) return 0;

    if (n == 1) { // trường hợp đặc biệt
        cout << "1\n";
        return 0;
    }

    bool first = true;
    for (long long i = 2; i * i <= n; ++i) {
        while (n % i == 0) {
            if (!first) cout << "*";
            cout << i;
            n /= i;
            first = false;
        }
    }
    if (n > 1) {
        if (!first) cout << "*";
        cout << n;
    }
    cout << "\n";
    return 0;
}
