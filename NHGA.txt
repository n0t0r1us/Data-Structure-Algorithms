#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;
    long long *a = new long long[n];
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    sort(a, a + n);

    long long min_dist = LLONG_MAX;
    for (int i = 1; i < n; i++) {
        long long diff = a[i] - a[i - 1];
        if (diff < min_dist) {
            min_dist = diff;
        }
    }

    cout << min_dist << "\n";
    delete[] a;
    return 0;
}
