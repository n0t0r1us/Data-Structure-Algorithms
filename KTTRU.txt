#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    string S;
    cin >> S;
    int len = S.size();
    int Q;
    cin >> Q;
    while (Q--) {
        long long a, b;
        cin >> a >> b;
        // Chuyển về chỉ số 0-based
        int idx1 = (int)((a - 1) % len);
        int idx2 = (int)((b - 1) % len);
        if (S[idx1] == S[idx2])
            cout << "Yes\n";
        else
            cout << "No\n";
    }
    return 0;
}
