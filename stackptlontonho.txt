#include <iostream>
#include <stack>
using namespace std;

void phanTichThuaSo(int n) {
    stack<int> st;
    int i = 2;

    while (n > 1 && i * i <= n) {
        while (n % i == 0) {
            st.push(i);
            n /= i;
        }
        i++;
    }

    if (n > 1) st.push(n); // còn một thừa số nguyên tố lớn hơn căn n

    cout << " = ";
    cout << st.top(); st.pop(); // in số đầu tiên
    while (!st.empty()) {
        cout << "*" << st.top();
        st.pop();
    }
    cout << endl;
}

int main() {
    int n;
    cin >> n;
    cout << n;
    phanTichThuaSo(n);
    return 0;
}
