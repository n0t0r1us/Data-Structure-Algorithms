#include <iostream>
using namespace std;

int n;
int a[1005][1005];
int dp[1005][1005];

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= n; ++j)
            cin >> a[i][j];

    int res = 0;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (a[i][j] == 0) {
                dp[i][j] = 1 + min(dp[i-1][j], min(dp[i][j-1], dp[i-1][j-1]));
                if (dp[i][j] > res)
                    res = dp[i][j];
            } else {
                dp[i][j] = 0;
            }
        }
    }

    cout << res << endl;
    return 0;
}
