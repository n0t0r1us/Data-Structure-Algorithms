#include <iostream>
#include <string>
#include <cstring>
using namespace std;

int main() {
    string s;
    // Read whole line including spaces
    if (!std::getline(cin, s)) return 0;
    bool seen[256];
    memset(seen, 0, sizeof(seen));
    int cnt = 0;
    for (size_t i = 0; i < s.size(); ++i) {
        unsigned char c = (unsigned char)s[i];
        if (!seen[c]) {
            seen[c] = true;
            ++cnt;
        }
    }
    cout << cnt << endl;
    return 0;
}
