#include <bits/stdc++.h>
using namespace std;

long long countZero(long long x) {
    long long cnt = 0;
    while (x > 0) {
        x /= 5;
        cnt += x;
    }
    return cnt;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int T;
    cin >> T;
    while (T--) {
        long long N;
        cin >> N;
        long long l = 1, r = N * 5, ans = -1;
        while (l <= r) {
            long long mid = (l + r) / 2;
            if (countZero(mid) >= N) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        cout << ans << "\n";
    }
    return 0;
}
