#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int m, n;
    cin >> m >> n;
    int a[n];
    for (int i = 0; i < n; ++i)
        cin >> a[i];

    sort(a, a + n);

    int res = 1000000000; // lớn hơn max giá trị có thể
    for (int i = 0; i + m - 1 < n; ++i) {
        int diff = a[i + m - 1] - a[i];
        if (diff < res) res = diff;
    }

    cout << res << "\n";
    return 0;
}
