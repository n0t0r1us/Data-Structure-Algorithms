#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    int mid = n / 2;

    for (int i = 0; i < n; i++) {
        int delta = abs(mid - i);

        for (int j = 0; j < delta; j++) cout << '*';
        for (int j = 0; j < n - 2 * delta; j++) cout << 'D';
        for (int j = 0; j < delta; j++) cout << '*';

        cout << endl;
    }

    return 0;
}
