#include <iostream>
#include <iomanip> // để setprecision
using namespace std;

int main() {
    int n;
    cin >> n;

    long long total = 0;
    for (int i = 0; i < n; ++i) {
        int quantity, price;
        cin >> quantity >> price;
        total += quantity * price;
    }

    double discount = 0.0;
    if (total >= 1000000)
        discount = 0.2;
    else if (total >= 500000)
        discount = 0.1;

    double final_amount = total * (1.0 - discount);

    cout << fixed << setprecision(2) << final_amount << endl;
    return 0;
}
