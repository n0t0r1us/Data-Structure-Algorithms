#include <iostream>
#include <cstdio>
#include <map>
using namespace std;

int main() {
    int k;
    scanf("%d", &k);

    map<long long, int> cnt;
    long long t;

    for (int i = 0; i < k; ++i) {
        scanf("%lld", &t);
        cnt[t]++;
    }

    long long total = 0;
    for (map<long long, int>::iterator it = cnt.begin(); it != cnt.end(); ++it) {
        int p = it->second;
        if (p <= 5) total += 100;
        else total += 100 + (p - 5);
    }

    printf("%lld\n", total);
    return 0;
}
