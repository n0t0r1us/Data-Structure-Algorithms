#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int n;
    if (!(cin >> n)) return 0;
    int sz = 2 * n + 1;
    int center = n;
    for (int i = 0; i < sz; ++i) {
        for (int j = 0; j < sz; ++j) {
            int d = abs(i - center) + abs(j - center);
            if (d % 3 == 0) cout << 'b';
            else cout << 'w';
        }
        if (i + 1 < sz) cout << '\n';
    }
    return 0;
}
