#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll n, m, k;

ll countLessEqual(ll x) {
    ll cnt = 0;
    for (ll i = 1; i <= n; i++) {
        cnt += min(m, x / i);
    }
    return cnt;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cin >> n >> m >> k;
    ll l = 1, r = n * m, ans = -1;
    while (l <= r) {
        ll mid = (l + r) / 2;
        if (countLessEqual(mid) >= k) {
            ans = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    cout << ans << "\n";
    return 0;
}
