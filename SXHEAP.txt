#include <iostream>
#include <vector>
using namespace std;
void heapify(vector<int>& a, int n, int i){
	int lonNhat = i;
	int l = 2*i + 1;
	int r = 2*i + 2;
	if(l < n && a[l] > a[lonNhat]) lonNhat = l;
	if(r < n && a[r] > a[lonNhat]) lonNhat = r;
	if(lonNhat != i){
		swap(a[i], a[lonNhat]);
		heapify(a, n, lonNhat);
	}
}
void printHeap(const vector<int> & a, int size){
	for(int i =0; i< size; i++)
	cout << a[i] << " ";
	cout << endl;
}
int main(){
	int n;
	cin >> n;
	vector<int> a(n);
	for(int i = 0; i< n; i++) cin >> a[i];
	for(int i = n /2 -1; i>= 0; i--)
	heapify(a, n, i);
	printHeap(a, n);
	for(int i = n - 1; i>0; i--){
		swap(a[0], a[i]);
		heapify(a, i, 0);
		printHeap(a, i);
	}
	return 0;
}
