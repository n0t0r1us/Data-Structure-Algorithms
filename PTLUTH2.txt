#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> res;
    for (int i = 17; i >= 0; --i) {
        int pow2 = 1 << i;
        if (n >= pow2) {
            res.push_back(pow2);
            n -= pow2;
        }
    }
    for (size_t i = 0; i < res.size(); ++i) {
        if (i) cout << "+";
        cout << res[i];
    }
    cout << endl;
    return 0;
}
