#include <iostream>
#include <queue>
using namespace std;

typedef long long ll;

int main() {
    int n;
    cin >> n;
    priority_queue<ll, vector<ll>, greater<ll> > pq; // min-heap

    for (int i = 0; i < n; ++i) {
        ll x;
        cin >> x;
        pq.push(x);
    }

    ll total_cost = 0;
    while (pq.size() > 1) {
        ll first = pq.top(); pq.pop();
        ll second = pq.top(); pq.pop();
        ll merged = first + second;
        total_cost += merged;
        pq.push(merged);
    }

    cout << total_cost << endl;
    return 0;
}
