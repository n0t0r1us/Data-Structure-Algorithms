#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;

    int a[10];
    for (int i = 0; i < n; ++i) {
        a[i] = i + 1;
    }

    int fact = 1;
    for (int i = 2; i <= n; ++i) fact *= i;
    cout << fact << endl;

    sort(a, a + n);
    do {
        for (int i = 0; i < n; ++i) {
            cout << a[i];
        }
        cout << '\n';
    } while (next_permutation(a, a + n));

    return 0;
}
