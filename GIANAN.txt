#include <cstdio>
#include <algorithm>

int a[105][105];
int dp[105][105];

int main() {
    int m, n;
    scanf("%d %d", &m, &n);

    for (int i = 1; i <= m; ++i)
        for (int j = 1; j <= n; ++j)
            scanf("%d", &a[i][j]);

    // Bắt đầu từ (1,1)
    for (int i = 1; i <= m; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (i == 1 && j == 1)
                dp[i][j] = 0; // ô (1,1) không có thức ăn
            else
                dp[i][j] = a[i][j] + std::max(dp[i-1][j], dp[i][j-1]);
        }
    }

    printf("%d\n", dp[m][n]);
    return 0;
}
