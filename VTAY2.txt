#include <iostream>
#include <string>
using namespace std;

int main() {
    string S;
    if (!(cin >> S)) return 0;
    int n = (int)S.size();
    string best = S;
    for (int i = 1; i < n; ++i) {
        // rotation bắt đầu từ i
        string t = S.substr(i) + S.substr(0, i);
        if (t < best) best = t;
    }
    cout << best << "\n";
    return 0;
}
