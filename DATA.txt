#include <iostream>
#include <vector>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n;
    if (!(cin >> n)) return 0;
    vector<long long> a(n+1);
    for (int i = 1; i <= n; ++i) cin >> a[i];

    int cnt = 0;
    int pos = -1; // position j where a[j] >= a[j+1] (with a[n+1]=a[1])
    for (int j = 1; j <= n; ++j) {
        int nxt = (j == n ? 1 : j+1);
        if (a[j] >= a[nxt]) {
            ++cnt;
            pos = j;
        }
    }

    if (cnt == 1) {
        int i = (pos == n ? 1 : pos + 1);
        if (i >= 2 && i <= n) cout << i << "\n";
        else cout << -1 << "\n";
    } else {
        cout << -1 << "\n";
    }
    return 0;
}
