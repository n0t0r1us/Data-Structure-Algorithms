#include <iostream>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int T;
    cin >> T;
    while (T--) {
        unsigned int N;
        cin >> N;

        if (N == 0) {
            // Với N=0, số phủ định là 1 (bit đơn)
            cout << 1 << "\n";
            continue;
        }

        int bits = 0;
        unsigned int temp = N;
        while (temp > 0) {
            bits++;
            temp >>= 1;
        }

        unsigned int mask = (1u << bits) - 1;
        cout << (N ^ mask) << "\n";
    }

    return 0;
}
