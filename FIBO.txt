#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    long long f1 = 1, f2 = 1;

    for (int i = 1; i <= n; ++i) {
        if (i == 1 || i == 2)
            cout << "1";
        else {
            long long fn = f1 + f2;
            cout << fn;
            f1 = f2;
            f2 = fn;
        }

        if (i < n) cout << " "; // In dấu cách giữa các số
    }

    cout << endl;
    return 0;
}
