#include <iostream>
#include <stack>
#include <sstream>
#include <string>
#include <cmath>
using namespace std;

int tinhHauTo(string expr) {
    stack<int> st;
    stringstream ss(expr);
    string token;

    while (ss >> token) {
        if (isdigit(token[0]) || (token.length() > 1 && token[0] == '-')) {
            st.push(stoi(token));
        } else {
            int b = st.top(); st.pop();
            int a = st.top(); st.pop();
            int res;

            if (token == "+") res = a + b;
            else if (token == "-") res = a - b;
            else if (token == "*") res = a * b;
            else if (token == "/") res = a / b;
            else if (token == "^") res = pow(a, b);
            else {
                cout << "Toan tu khong hop le: " << token << endl;
                return 0;
            }

            st.push(res);
        }
    }

    return st.top();
}

int main() {
    string expr;
    getline(cin, expr);
    cout << tinhHauTo(expr) << endl;
    return 0;
}
