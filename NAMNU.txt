#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main() {
    int M, N;
    cin >> M >> N;
    string s;
    cin >> s;
    int len = M + N;
    int countMM = 0, countFF = 0;

    for (int i = 0; i < len; i++) {
        char a = s[i];
        char b = s[(i + 1) % len]; // vòng tròn, nên %len
        if (a == '1' && b == '1') countMM++;
        else if (a == '0' && b == '0') countFF++;
    }

    cout << abs(countMM - countFF) << endl;
    return 0;
}
