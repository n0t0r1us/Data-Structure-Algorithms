#include <iostream>
#include <vector>
using namespace std;

struct Program {
    int start, end;
};

int main() {
    int n;
    cin >> n;
    vector<Program> a(n);

    for (int i = 0; i < n; ++i) {
        cin >> a[i].start >> a[i].end;
    }

    int count = 0;

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            // Nếu thời gian chiếu chồng lấn ⇒ không phù hợp
            if (a[i].start < a[j].end && a[j].start < a[i].end) {
                count++;
            }
        }
    }

    cout << count << endl;

    return 0;
}
