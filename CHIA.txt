#include <iostream>
#include <string>
using namespace std;

string addInt(const string &s, int x) {
    string res = s;
    int carry = x, i = res.size() - 1;
    while (i >= 0 && carry > 0) {
        int sum = (res[i] - '0') + carry;
        res[i] = char(sum % 10 + '0');
        carry = sum / 10;
        i--;
    }
    if (carry > 0) res.insert(res.begin(), char(carry + '0'));
    return res;
}

int modInt(const string &s, int m) {
    int r = 0;
    for (int i = 0; i < (int)s.size(); i++)
        r = (r * 10 + (s[i] - '0')) % m;
    return r;
}

int main() {
    int n;
    cin >> n;
    if (n == 1) {
        cout << -1;
        return 0;
    }
    string num = "1";
    num.append(n - 1, '0');
    int rem = modInt(num, 21);
    if (rem != 0) num = addInt(num, 21 - rem);
    if ((int)num.size() > n) cout << -1;
    else cout << num;
    return 0;
}
