#include <iostream>
using namespace std;

typedef unsigned long long ull;

const int MAX_BITS = 64;

ull basis[MAX_BITS]; // cơ sở XOR theo từng bit

void insert_basis(ull x) {
    for (int i = MAX_BITS - 1; i >= 0; --i) {
        if ((x >> i) & 1) { // nếu bit i của x = 1
            if (!basis[i]) {
                basis[i] = x;
                return;
            }
            x ^= basis[i];
        }
    }
}

ull get_max_xor() {
    ull res = 0;
    for (int i = MAX_BITS - 1; i >= 0; --i) {
        if ((res ^ basis[i]) > res)
            res ^= basis[i];
    }
    return res;
}

int main() {
    int n;
    cin >> n;
    ull x;
    for (int i = 0; i < n; ++i) {
        cin >> x;
        insert_basis(x);
    }

    cout << get_max_xor() << endl;
    return 0;
}
