#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int a[25];
    for (int i = 0; i < n; ++i)
        cin >> a[i];

    int pivot = a[n - 1];
    int i = 0;
    for (int j = 0; j < n - 1; ++j) {
        if (a[j] <= pivot) {
            swap(a[i], a[j]);
            i++;
        }
    }
    swap(a[i], a[n - 1]); // đưa pivot về giữa

    // In kết quả
    for (int k = 0; k < n; ++k) {
        if (k == i)
            cout << "[" << a[k] << "]";
        else
            cout << a[k];
        if (k != n - 1)
            cout << " ";
    }
    cout << endl;

    return 0;
}
