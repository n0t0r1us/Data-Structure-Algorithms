#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int N, M;
    cin >> N >> M;
    vector<long long> a(N), b(N);
    for (int i = 0; i < N; i++) {
        cin >> a[i] >> b[i];
    }
    while (M--) {
        long long x, y, t;
        cin >> x >> y >> t;
        int count = 0;
        for (int i = 0; i < N; i++) {
            long long d = b[i] - a[i];
            if (d == 0) { // xe không di chuyển
                if (a[i] >= x && a[i] <= y) count++;
                continue;
            }
            long long cycle = 2 * d;
            long long pos_in_cycle = t % cycle;
            long long pos;
            if (pos_in_cycle <= d) {
                pos = a[i] + pos_in_cycle;
            } else {
                pos = b[i] - (pos_in_cycle - d);
            }
            if (pos >= x && pos <= y) count++;
        }
        cout << count << "\n";
    }
    return 0;
}
