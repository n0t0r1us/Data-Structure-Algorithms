#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

int main() {
    int n;
    cin >> n;

    vector<int> nam_up, nam_down;
    vector<int> nu_up, nu_down;

    for (int i = 0; i < n; ++i) {
        int h;
        cin >> h;
        if (h > 0) nam_up.push_back(h);
        else nam_down.push_back(-h);
    }

    for (int i = 0; i < n; ++i) {
        int h;
        cin >> h;
        if (h > 0) nu_up.push_back(h);
        else nu_down.push_back(-h);
    }

    sort(nam_up.begin(), nam_up.end());
    sort(nam_down.begin(), nam_down.end());
    sort(nu_up.begin(), nu_up.end());
    sort(nu_down.begin(), nu_down.end());

    int count = 0;

    // Nam muốn nhảy với nữ cao hơn ↔ Nữ muốn nhảy với nam thấp hơn
    int i = 0, j = 0;
    while (i < nam_up.size() && j < nu_down.size()) {
        if (nam_up[i] < nu_down[j]) {
            ++count;
            ++i;
            ++j;
        } else {
            ++j;
        }
    }

    // Nam muốn nhảy với nữ thấp hơn ↔ Nữ muốn nhảy với nam cao hơn
    i = 0; j = 0;
    while (i < nam_down.size() && j < nu_up.size()) {
        if (nam_down[i] > nu_up[j]) {
            ++count;
            ++i;
            ++j;
        } else {
            ++i;
        }
    }

    cout << count << endl;
    return 0;
}
