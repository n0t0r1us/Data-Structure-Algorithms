#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string n, k;
    cin >> n >> k;

    while (n.length() < k.length()) n = '0' + n;
    while (k.length() < n.length()) k = '0' + k;

    string res = "";
    for (size_t i = 0; i < n.length(); ++i) {
        int a = n[i] - '0';
        int b = k[i] - '0';
        res += (char)((a + b) % 10 + '0');
    }

    // Loại bỏ số 0 đầu nếu có
    int pos = 0;
    while (pos < res.length() - 1 && res[pos] == '0') ++pos;
    cout << res.substr(pos) << endl;

    return 0;
}
