#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n;
    cin >> n;
    vector<int> a(n);
    int sum = 0;
    for(int i=0;i<n;i++){
        cin >> a[i];
        sum += a[i];
    }
    int ans = 1;
    for(int parts = n; parts >= 1; parts--){
        if(sum % parts != 0) continue;
        int target = sum / parts;
        int cnt = 0, cur = 0;
        bool ok = true;
        for(int i=0;i<n;i++){
            cur += a[i];
            if(cur > target){ ok = false; break; }
            if(cur == target){ cnt++; cur = 0; }
        }
        if(ok && cnt == parts){ ans = parts; break; }
    }
    cout << ans << "\n";
    return 0;
}
