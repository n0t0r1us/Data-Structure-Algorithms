#include <stdio.h>

void printArray(int arr[], int n, int inserted_index) {
    for (int i = 0; i < n; i++) {
        if (i == inserted_index) {
            printf("[%d] ", arr[i]);
        } else {
            printf("%d ", arr[i]);
        }
    }
    printf("\n");
}

void insertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i];
        int j = i - 1;
        
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
        
        printArray(arr, n, j + 1);
    }
}

int main() {
    int n;
    scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    insertionSort(arr, n);
    
    return 0;
}
