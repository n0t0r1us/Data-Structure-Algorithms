#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef pair<int, int> Interval;

int main() {
    int n;
    cin >> n;

    vector<Interval> segs(n);

    for (int i = 0; i < n; ++i) {
        cin >> segs[i].first >> segs[i].second;
    }

    // Sắp xếp các đoạn theo điểm bắt đầu
    sort(segs.begin(), segs.end());

    long long total = 0;
    int current_start = segs[0].first;
    int current_end = segs[0].second;

    for (int i = 1; i < n; ++i) {
        int s = segs[i].first;
        int e = segs[i].second;

        if (s > current_end) {
            // Không giao nhau: cộng chiều dài đoạn trước vào tổng
            total += current_end - current_start;
            current_start = s;
            current_end = e;
        } else {
            // Giao nhau: gộp đoạn lại
            if (e > current_end) {
                current_end = e;
            }
        }
    }

    // Cộng đoạn cuối cùng
    total += current_end - current_start;

    cout << total << endl;

    return 0;
}
