#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    long long a, b, c;
    while (cin >> a >> b >> c) {
        if (a == 0 && b == 0 && c == 0) break;
        if (b - a == c - b) {
            cout << "AP " << c + (b - a) << "\n";
        } else {
            cout << "GP " << c * (b / a) << "\n";
        }
    }
    return 0;
}
