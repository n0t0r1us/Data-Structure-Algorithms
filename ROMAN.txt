#include <iostream>
#include <string>
using namespace std;

int romanCharToInt(char c) {
    if (c == 'I') return 1;
    if (c == 'V') return 5;
    if (c == 'X') return 10;
    if (c == 'L') return 50;
    if (c == 'C') return 100;
    if (c == 'D') return 500;
    if (c == 'M') return 1000;
    return 0;
}

int main() {
    string s;
    cin >> s;
    int result = 0;
    int n = s.length();

    for (int i = 0; i < n; ++i) {
        int val1 = romanCharToInt(s[i]);
        int val2 = (i + 1 < n) ? romanCharToInt(s[i + 1]) : 0;

        if (val1 < val2) {
            result -= val1;
        } else {
            result += val1;
        }
    }

    cout << result << endl;
    return 0;
}
