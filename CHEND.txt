#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

void insertHead(Node*& head, int value) {
    Node* newNode = new Node;
    newNode->data = value;
    newNode->next = head;
    head = newNode;
}

void printList(Node* head) {
    Node* current = head;
    while (current != NULL) {  // dùng NULL thay vì nullptr
        cout << current->data << " ";
        current = current->next;
    }
}

int main() {
    int n;
    cin >> n;

    Node* head = NULL; // dùng NULL thay vì nullptr

    for (int i = 0; i < n; ++i) {
        int value;
        cin >> value;
        insertHead(head, value);
    }

    printList(head);

    return 0;
}
