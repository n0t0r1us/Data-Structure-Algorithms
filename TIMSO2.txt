#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    long long M;
    cin >> M;

    if (M == 0) {
        cout << "10\n";
        return 0;
    }
    if (M == 1) {
        cout << "1\n";
        return 0;
    }

    vector<int> digits;
    for (int d = 9; d >= 2; --d) {
        while (M % d == 0) {
            digits.push_back(d);
            M /= d;
        }
    }

    if (M > 1) {
        cout << "-1\n";
        return 0;
    }

    sort(digits.begin(), digits.end());
    for (int i = 0; i < digits.size(); ++i)
        cout << digits[i];
    cout << endl;

    return 0;
}
