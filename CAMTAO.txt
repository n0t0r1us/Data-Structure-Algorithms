#include <iostream>
#include <string>
using namespace std;

int main() {
    int n;
    cin >> n;
    string s;
    cin >> s;
    int diff1 = 0, diff2 = 0;
    for (int i = 0; i < 2 * n; ++i) {
        char expected1 = (i % 2 == 0) ? '0' : '1';
        char expected2 = (i % 2 == 0) ? '1' : '0';
        if (s[i] != expected1) ++diff1;
        if (s[i] != expected2) ++diff2;
    }
    cout << (diff1 < diff2 ? diff1 : diff2) << endl;
    return 0;
}
