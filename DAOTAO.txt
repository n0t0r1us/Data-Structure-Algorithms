#include <iostream>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    int c[101];
    int total = 0;
    int f_max = 0;
    int count_max = 0;

    for (int i = 0; i < n; ++i) {
        cin >> c[i];
        total += c[i];

        if (c[i] > f_max) {
            f_max = c[i];
            count_max = 1;
        } else if (c[i] == f_max) {
            count_max++;
        }
    }

    int res = (f_max - 1) * m + count_max;
    if (res < total) res = total;

    cout << res << endl;
    return 0;
}
