#include <iostream>
#include <vector>
using namespace std;

int k, n;
vector<string> result;

void Try(string current, int depth) {
    if (depth == n) {
        result.push_back(current);
        return;
    }

    for (int i = 1; i <= k; ++i) {
        Try(current + char(i + '0'), depth + 1);
    }
}

int main() {
    cin >> k >> n;
    Try("", 0);

    cout << result.size() << endl;
    for (int i = 0; i < (int)result.size(); ++i) {
        cout << result[i] << endl;
    }

    return 0;
}
