#include <iostream>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    int steps = n * m;
    
    long long f[steps + 2]; // tránh tràn mảng
    f[0] = 1;
    f[1] = 1;
    
    for (int i = 2; i <= steps; i++) {
        f[i] = f[i - 1] + f[i - 2];
    }

    cout << f[steps] << endl;
    return 0;
}
