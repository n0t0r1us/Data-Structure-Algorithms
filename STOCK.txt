#include <iostream>
using namespace std;

int main() {
    int N;
    long long W;
    cin >> N >> W;
    int p[105];
    for (int i = 0; i < N; ++i) {
        cin >> p[i];
    }

    long long max_profit = 0;

    for (int i = 0; i < N; ++i) {
        int buy_price = p[i];
        long long shares = W / buy_price;
        for (int j = i + 1; j < N; ++j) {
            int sell_price = p[j];
            long long profit = shares * (sell_price - buy_price);
            if (profit > max_profit)
                max_profit = profit;
        }
    }

    cout << max_profit;
    return 0;
}
