#include <bits/stdc++.h>
using namespace std;

long long gcd_ll(long long a, long long b) {
    while (b) {
        long long t = a % b;
        a = b;
        b = t;
    }
    return a;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int T;
    cin >> T;
    while (T--) {
        int N;
        cin >> N;
        vector<long long> A(N);
        for (int i = 0; i < N; i++) cin >> A[i];
        vector<long long> B(N + 1);
        B[0] = A[0];
        for (int i = 1; i < N; i++) {
            long long g = gcd_ll(A[i - 1], A[i]);
            B[i] = (A[i - 1] / g) * A[i];
        }
        B[N] = A[N - 1];
        for (int i = 0; i <= N; i++) {
            cout << B[i] << (i == N ? '\n' : ' ');
        }
    }
    return 0;
}
