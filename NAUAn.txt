#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int n;
    cin >> n;
    vector<long long> a(n), prefix(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        prefix[i] = a[i] + (i ? prefix[i-1] : 0);
    }
    int m;
    cin >> m;
    while (m--) {
        long long p;
        cin >> p;
        int idx = lower_bound(prefix.begin(), prefix.end(), p) - prefix.begin();
        cout << idx + 1 << " ";
    }
    return 0;
}
