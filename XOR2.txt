#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int T;
    cin >> T;
    while (T--) {
        int N;
        cin >> N;
        vector<int> a(N), pref(N + 1, 0);
        for (int i = 0; i < N; i++) {
            cin >> a[i];
            pref[i + 1] = pref[i] ^ a[i];
        }
        int Q;
        cin >> Q;
        while (Q--) {
            int l, r;
            cin >> l >> r;
            cout << (pref[r + 1] ^ pref[l]) << "\n";
        }
    }
    return 0;
}
