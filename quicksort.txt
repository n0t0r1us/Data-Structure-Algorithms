#include <iostream>
using namespace std;

void quickSort(int a[], int left, int right) {
    if (left >= right) return;

    int pivot = a[(left + right) / 2];
    int i = left, j = right;

    while (i <= j) {
        while (a[i] < pivot) i++;
        while (a[j] > pivot) j--;
        if (i <= j) {
            int tmp = a[i];
            a[i] = a[j];
            a[j] = tmp;
            i++;
            j--;
        }
    }

    quickSort(a, left, j);
    quickSort(a, i, right);
}

int main() {
    int n;
    cin >> n;

    int a[100];
    for (int i = 0; i < n; ++i)
        cin >> a[i];

    quickSort(a, 0, n - 1);

    for (int i = 0; i < n; ++i)
        cout << a[i] << " ";

    return 0;
}
