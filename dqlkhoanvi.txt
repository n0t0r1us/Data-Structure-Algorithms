#include <iostream>
using namespace std;

int n, a[10];
bool used[10];
int total = 0;

void Try(int i) {
    for (int j = 1; j <= n; j++) {
        if (!used[j]) {
            a[i] = j;
            used[j] = true;
            if (i == n) {
                total++;
                for (int k = 1; k <= n; k++) cout << a[k];
                cout << endl;
            } else {
                Try(i + 1);
            }
            used[j] = false;
        }
    }
}

int main() {
    cin >> n;
    Try(1);
    cout << total << endl;
    return 0;
}
