#include <iostream>
using namespace std;

typedef long long ll;

ll findSegment(ll k) {
    ll low = 1, high = 2e9, mid, sum;
    while (low < high) {
        mid = (low + high) / 2;
        sum = mid * (mid + 1) / 2;
        if (sum >= k)
            high = mid;
        else
            low = mid + 1;
    }
    return low;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int T;
    cin >> T;
    while (T--) {
        ll k;
        cin >> k;
        ll seg = findSegment(k);
        ll prev = (seg - 1) * seg / 2;
        ll pos = k - prev;
        cout << (pos % 10 == 0 ? 0 : pos % 10) << "\n";
    }
    return 0;
}
