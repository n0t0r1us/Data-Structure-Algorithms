#include <iostream>
#include <string>
using namespace std;

int rank_of(char c) {
    if (c == 'T') return 10;
    if (c == 'J') return 11;
    if (c == 'Q') return 12;
    if (c == 'K') return 13;
    if (c == 'A') return 14;
    if (c == '2') return 15;
    // digits '3'..'9'
    return c - '0';
}

int main() {
    int cnt[16];
    for (int i = 0; i < 16; ++i) cnt[i] = 0;
    string s;
    for (int i = 0; i < 13; ++i) {
        if (!(cin >> s)) return 0;
        // s[0] is rank char, s[1] is suit
        int r = rank_of(s[0]);
        if (r >= 3 && r <= 15) cnt[r]++;
    }

    // Condition 1: full 13 distinct ranks
    int distinct = 0;
    for (int r = 3; r <= 15; ++r) if (cnt[r] > 0) distinct++;
    if (distinct == 13) {
        cout << "YES\n";
        return 0;
    }

    // Condition 2: four 2's
    if (cnt[15] == 4) {
        cout << "YES\n";
        return 0;
    }

    // Condition 3: five consecutive pairs
    for (int start = 3; start <= 11; ++start) {
        bool ok = true;
        for (int r = start; r <= start + 4; ++r) {
            if (cnt[r] < 2) { ok = false; break; }
        }
        if (ok) {
            cout << "YES\n";
            return 0;
        }
    }

    // Condition 4: six arbitrary pairs (four-of-a-kind counts as 2 pairs)
    int pairs = 0;
    for (int r = 3; r <= 15; ++r) pairs += cnt[r] / 2;
    if (pairs >= 6) {
        cout << "YES\n";
        return 0;
    }

    cout << "NO\n";
    return 0;
}
