#include <iostream>
#include <cstring>
using namespace std;

int main() {
    char s[105];
    cin >> s;

    int count = 0;
    int n = strlen(s);
    bool valid = true;

    for (int i = 0; i < n; i++) {
        if (s[i] == '(') {
            count++;
        } else if (s[i] == ')') {
            count--;
            if (count < 0) {
                valid = false;
                break;
            }
        }
    }

    if (count != 0) valid = false;

    if (valid)
        cout << "YES" << endl;
    else
        cout << "NO" << endl;

    return 0;
}
