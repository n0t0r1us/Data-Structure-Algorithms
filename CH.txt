#include <iostream>
using namespace std;

int n, k;
int x[100];
bool used[100];

void Try(int i) {
    for (int j = 1; j <= n; ++j) {
        if (!used[j]) {
            x[i] = j;
            used[j] = true;
            if (i == k) {
                for (int l = 1; l <= k; ++l)
                    cout << x[l] << " ";
                cout << endl;
            } else {
                Try(i + 1);
            }
            used[j] = false;
        }
    }
}

int main() {
    cin >> n >> k;
    Try(1);
    return 0;
}
