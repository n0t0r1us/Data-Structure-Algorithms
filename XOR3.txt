#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int T;
    cin >> T;
    while (T--) {
        int N;
        cin >> N;
        vector<long long> A(N);
        for (int i = 0; i < N; i++) cin >> A[i];
        long long ans = 0;
        for (int i = 0; i < N; i++) {
            long long cnt = (long long)(i + 1) * (N - i);
            if (cnt % 2) ans ^= A[i];
        }
        cout << ans << "\n";
    }
    return 0;
}
