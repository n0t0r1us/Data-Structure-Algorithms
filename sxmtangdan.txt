#include <iostream>
using namespace std;

int main() {
    int a[101], n, x;
    
    cin >> n;
    for (int i = 0; i < n; ++i)
        cin >> a[i];
    cin >> x;

    // Tìm vị trí chèn
    int pos = n;
    for (int i = 0; i < n; ++i) {
        if (x < a[i]) {
            pos = i;
            break;
        }
    }

    // Dời các phần tử sang phải để chừa chỗ chèn
    for (int i = n; i > pos; --i)
        a[i] = a[i - 1];

    a[pos] = x;
    ++n;

    // In ra mảng sau khi chèn
    for (int i = 0; i < n; ++i)
        cout << a[i] << " ";

    return 0;
}
