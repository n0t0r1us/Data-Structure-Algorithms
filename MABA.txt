#include <iostream>
using namespace std;

int main() {
    int n, m, k;
    cin >> n >> m >> k;

    int count[101][101] = {0}; // lưu số đường hoàn thành mỗi ô
    for (int _=0; _<k; _++) {
        char t; int v;
        cin >> t >> v;
        if (t == 'H') {
            // Đường ngang H v
            if (v > 0) {
                for (int j = 0; j < m; j++) {
                    count[v-1][j]++;
                }
            }
            if (v < n) {
                for (int j = 0; j < m; j++) {
                    count[v][j]++;
                }
            }
        } else {
            // Đường dọc V v
            if (v > 0) {
                for (int i = 0; i < n; i++) {
                    count[i][v-1]++;
                }
            }
            if (v < m) {
                for (int i = 0; i < n; i++) {
                    count[i][v]++;
                }
            }
        }
    }

    int a[5] = {0};
    for (int i=0; i<n; i++) {
        for (int j=0; j<m; j++) {
            int s = count[i][j];
            if (s>=0 && s<=4) a[s]++;
        }
    }

    for (int i=0; i<=4; i++) {
        cout << a[i];
        if (i<4) cout << " ";
    }
    cout << "\n";

    return 0;
}
